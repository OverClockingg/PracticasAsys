% Parámetro de frecuencia de corte
omega_c = 2;
 
% Definir las frecuencias de interés para la gráfica de Bode
omega = logspace(-1, 2, 500); % De 0.1 a 100 rad/s
 
% Definir las funciones de transferencia en términos de jω
G1 = @(omega) omega_c ./ (1i*omega + omega_c); % Sistema original
G2 = @(omega) omega_c ./ (1i*omega + (1 + 9) * omega_c); % k = 9
G3 = @(omega) omega_c ./ (1i*omega + (1 + 0.9) * omega_c); % k = 0.9
 
% Calcular la magnitud y la fase para cada frecuencia
magnitude_G1 = abs(G1(omega));
phase_G1 = angle(G1(omega)) * 180/pi; % Convertir a grados
 
magnitude_G2 = abs(G2(omega));
phase_G2 = angle(G2(omega)) * 180/pi; % Convertir a grados
 
magnitude_G3 = abs(G3(omega));
phase_G3 = angle(G3(omega)) * 180/pi; % Convertir a grados
 
% Crear una nueva figura para las gráficas de Bode
figure;
% Subplot para la magnitud
subplot(2, 1, 1);
semilogx(omega, 20*log10(magnitude_G1), 'b'); % Magnitud en dB para G1 (azul)
hold on;
semilogx(omega, 20*log10(magnitude_G2), 'r'); % Magnitud en dB para G2 (rojo)
semilogx(omega, 20*log10(magnitude_G3), 'g'); % Magnitud en dB para G3 (verde)
title('Diagrama de Bode - Magnitud');
xlabel('Frecuencia (rad/s)');
ylabel('Magnitud (dB)');
legend('Sistema Original G(s)', 'Retroalimentación k=9', 'Retroalimentación k=0.9');
grid on;
 
% Subplot para la fase
subplot(2, 1, 2);
semilogx(omega, phase_G1, 'b'); % Fase en grados para G1 (azul)
hold on;
semilogx(omega, phase_G2, 'r'); % Fase en grados para G2 (rojo)
semilogx(omega, phase_G3, 'g'); % Fase en grados para G3 (verde)
title('Diagrama de Bode - Fase');
xlabel('Frecuencia (rad/s)');
ylabel('Fase (grados)');
legend('Sistema Original G(s)', 'Retroalimentación k=9', 'Retroalimentación k=0.9');
grid on;
 
% Ajustar ejes y la apariencia
set(gcf, 'Position', [100, 100, 800, 600]); % Ajustar el tamaño de la figura
