function ecd1(CS,CE,ciy,cix,fentrada,n0)
tam=size(CS); %Coeficientes de salida
tami=size(CE); %Coeficientes de entrada
cam=size(ciy); % Tamaño condicones inciales
syms y(n) n z Y(z) x(n) X(z) Yy fp yf(n) xf(n);
syms edd edi
xi=fentrada;
edd=0;
edi=0;

Menu: Condiciones de retardo o adelanto a la señal

% condiciones auxiliares y[0], y[1],...,y[N-1]
% condiciones iniciales y[-1], y[-2],...,y[-n]
disp('------------------------------------------------------------------------')
disp('¿En necesario establecer condiciones de retardo a la señal? ')
disp('1-No *-condiciones auxiliares y[0], y[1],...,y[N-1] ')
disp('2-Si *-condiciones iniciales y[-1], y[-2],...,y[-n] ')
disp('------------------------------------------------------------------------')
condicion=input('Eliga alguna opcion: ');

Condicion de retardo en la señas( solo si aplica )

if condicion==2
for i=1:tam(end)
yd(i)=y(n+tam(end)-i-cam(end)); % Si la ecuacion necesita desplazar se recorre el numero de condiciones inciales ya que las necesitaremos susitituir despues
edd=edd+ CS(i)*ztrans(yd(i)) ;
end
else
for i=1:tam(end)
yd(i)=y(n+tam(end)-i);
edd=edd+ CS(i)*ztrans(yd(i));
end
end

Dezplasamiento de la señal y establecimiento de los coeficientes de la ecuacíon

disp('______Menu_______ ')
disp('1 *- 1 Dezplazado a la derecha ')
disp('2 *- 2 Dezplazado a la izquierda ')
menu=input('Ingrese hacia donde esta desplazada la entrada: ');
if menu == 1
if tami==1
for i=1:tami(end)
xd(i)=x(n-tami(end)+i-1);
edi=edi+ CE(i)*ztrans(xd(i)) %Transformada multiplicado por su coeficiente
end
else
for i=1:tami(end)
xd(i)=x(n-tami(end)+i);
edi=edi+ CE(i)*ztrans(xd(i)) %Transformada multiplicado por su coeficiente
end
end
end

Condicion para el dezplasamiento a la izquierda

if menu == 2
if tami==1
for i=1:tami(end)
xd(i)=x(n-tami(end)+i+1)
edi=edi+ CE(i)*ztrans(xd(i)) %Transformada multiplicado por su coeficiente
end
else
for i=1:tami(end)
xd(i)=x(n+tami(end)-i)
edi=edi+ CE(i)*ztrans(xd(i))%Transformada multiplicado por su coeficiente
end
end
end

Aplicación de transformada Z a ambos lados de la ecuación

disp('------------------------------------------------------------------------')
disp('APLICAMOS TRANSFORMADA ZETA ')
edd=subs(edd,ztrans(y(n),n,z), Y(z));
edi=subs(edi,ztrans(x(n),n,z), X(z));
pretty(edd)
disp('=')
pretty(edi)
disp('------------------------------------------------------------------------')

Función de transferencia y respuesta al impulso

disp('FUNCION DE TRANSFERENCIA ')
edd2=subs(edd,ztrans(yf(n),n,z), Y(z));
edi2=subs(edi,ztrans(xf(n),n,z), X(z));
pretty(edd2)
disp('=')
pretty(edi2)
disp('------------------------------------------------------------------------')
disp('SUPONIENDO TODAS LAS C.I IGUALA CERO')
for j=1:tami(end)
edi2=subs(edi2,x(tami(end)-1-j),0);
end
for j=1:tam(end)
edd2=subs(edd2,y(-1*j),0);
end
pretty(edd2)
disp('=')
pretty(edi2)
disp('------------------------------------------------------------------------')
disp('SUBSTITUIMOS LA ENTRADA POR LA DELTA DE KRONECKER')
disp('PARA OBTENER LA RESPUESTA AL IMPULSO')
edi2=subs(edi2,X(z), ztrans(kroneckerDelta(n)));
disp(' Z[zero-state response]')
pretty(edd2)
disp('=')
disp(' Z[input]')
pretty(edi2)
edd2=collect(edd2,Y(z));
edd2=subs(edd2,Y(z),Yy);
eq3=edd2==edi2;
disp('Y(z)=')
edd2=solve(eq3, Yy);
pretty(edd2)
disp('------------------------------------------------------------------------')
disp('Aplicamos transformada inversa, asi la solucion es')
disp('Respuesta al impulso =')
Yim(n)=iztrans(edd2);
pretty(Yim(n))
